    VALUES ?t { mi:TFG mi:TFM }
    ?work a ?t .
    OPTIONAL { ?work mi:year ?year. }
    OPTIONAL { ?work mi:language ?lang. }

    # Busca por términos relevantes usando el índice Lucene
    (?work ?score) text:query ("animaci* OR (vision* AND comput*)").

    FILTER ( 
        (BOUND(?year) && ?year >= "2005"^^xsd:gYear) &&
        (BOUND(?lang) && regex(str(?lang), "^spa", "i") )
    )
}

SELECT DISTINCT ?work ?title ?authorName ?score WHERE {
    VALUES ?t { mi:TFG mi:TFM }
    ?work a ?t .
    ?work mi:publisher ?dept .
    OPTIONAL { ?work mi:title ?title. }
    OPTIONAL { ?work mi:hasAuthor ?author. ?author mi:namePersona ?authorName. }
    OPTIONAL { ?work mi:contributor ?contrib . ?contrib mi:namePersona ?contribName. }

    # Filtrar por apellido López en autor/director
    FILTER (
        (BOUND(?authorName) && regex(?authorName, "L[oó]pez", "i")) ||
        (BOUND(?contribName) && regex(?contribName, "L[oó]pez", "i"))
    )

    # Ordenar por relevancia en torno a informática (consulta lucene)
    (?dept ?score) text:query ("informatic* OR \"departamento de informatica\"" ).

    OPTIONAL { ?work mi:description ?description. }
    FILTER ( !BOUND(?description) || regex(?description, "inform[áa]tica|departamento de inform[áa]tica", "i") )
}
ORDER BY DESC(?score)
LIMIT 100
